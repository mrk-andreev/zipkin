How to install helm chart with zipkin

  • Zipkin for postgres implementation https://github.com/mrk-andreev/zipkin/tree/1232-postgres
  • Chart source https://github.com/mrk-andreev/zipkin/tree/1232-postgres-chart 
  kubectl create namespace zipkin
  helm repo add zipkin https://mrkandreev.name/zipkin/charts
  helm repo update

  helm upgrade --install \
    zipkin zipkin/zipkin \
    --namespace zipkin \
    --version v1.0.0 \
    --set db.host=localhost \
    --set db.user=postgres \
    --set db.password=postgres \
    --set db.schema=public \
    --set db.database=postgres \
    --set db.maxConnections=10 \
    --set db.useSsl=false
